# sinusoidal.txt
# A script for the command line interface,
# configuring two carrier boards to calculate a sinusoidal each.

# Set alias for carrier
set-alias * carrier-one
set-alias * carrier-two

# Configure routing on first cluster
route --no-sync -- carrier-one/0 0 0 -1.0 1
route --no-sync -- carrier-one/0 1 1  1.0 0
# Configure initial condition
set-element-config carrier-one/0/M0/0 ic 0.42

# Configure routing on second cluster
route --no-sync -- carrier-two/0 0 0 -1.0 1
route --no-sync -- carrier-two/0 1 1  1.0 0
# Configure initial condition
set-element-config carrier-two/0/M0/0 ic 0.42

# Configure data acquisition for two channels and 100000 samples/second
set-daq -n 2 -r 100000 carrier-one/0/M0/0 carrier-one/0/M0/1 carrier-two/0/M0/0 carrier-two/0/M0/1

# Start run and save data
run --op-time 2560000 --output sinusoidal.dat
